CREATE TABLE CUSTOMERS(
CUSTOMERID NUMBER(3) PRIMARY KEY,
CUSTOMERNAME VARCHAR2(30),
CONTACTNAME VARCHAR2(30),
ADDRESS VARCHAR2(105),
CITY VARCHAR2(20),
POSTALCODE NUMBER(5),
COUNTRY VARCHAR2(15));
 
CREATE TABLE SUPPLIERS(
SUPPLIERID NUMBER(3) PRIMARY KEY,
SUPPLIERNAME VARCHAR2(30),
CONTACTNAME VARCHAR2(30),
ADDRESS VARCHAR2(75),
CITY VARCHAR2(20),
POSTALCODE NUMBER(5),
COUNTRY VARCHAR2(15));
